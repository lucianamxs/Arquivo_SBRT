1function x_new = vad_s(x,fs)
 2% a simple Voice Activity Detection program for speech processing
 3
 4if nargin <1
 5    error('no data ???');
 6end
 7if nargin <2
 8    fs = 8000;
 9end
10
11% an energy threshold is used to remove the silence parts of speech signal
12% the speech is assumed to be clean
13idx = find(x.^2>max(abs(x))^2/(10^(50/10))); % or 30db?
14diff = idx(2:end)-idx(1:end-1);
15thresh = 0.05*fs; % set threshold to 50ms
16idx2 = [idx(diff>thresh)'; idx(find(diff>thresh)+1)'];
17idx2 = idx2(:);
18
19if ~isempty(idx2)
20    if length(idx2)>2
21        x_new = x(idx(1):idx2(1));
22        for ii = 2:2:length(idx2)-2
23            x_new = [x_new ; x(idx2(ii):idx2(ii+1))];
24        end
25        x_new = [x_new ; x(idx2(end):idx(end))];
26    else
27        x_new = [x(idx(1):idx2(1)) ; x(idx2(1):idx(end))];
28    end
29else
30    x_new = x(idx(1):idx(end));
31end
